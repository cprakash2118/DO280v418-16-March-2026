 2-) TOken Based via OAuth

	$ oc login -u admin -p redhatocp https://api.ocp4.example.com:6443 
     	 
	Login to OCP cluster via newly created users (IdP=htpasswd)
	Step-1-# Create Database (via htpasswd)
	  $ htpasswd -c -b -B htpasswd-file user1 redhat 
	  $ htpasswd    -b -B htpasswd-file user2 redhat 
	  $ cat htpasswd
	  
	Step-2-# Inject to OCp cluster via secret (genric).
	  $ oc create secret generic secret-name --from-file httpaswd=htpasswd-file -n openshift-config 
	  $ oc get secret secret-name -n openshift-config
	  
	  Extract the secret.
	  oc extract secret/secret-name --to=. -n openshift-config  - download 
	  oc extract secret/secret-name --to=- -n openshift-config  - View without download.
	  
	Step-3-# Update to Oauth file.
	 $ oc get pods -n openshift-authentication 
	 
	Step-4-# Verify the login to OCP cluster.
	 $ oc get oauth -o yaml
